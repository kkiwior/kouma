{{- else if eq .Values.dashboard.service.type "NodePort" }}

Access Kouma via NodePort:
  export NODE_PORT=$(kubectl get svc {{ $fullName }}-dashboard -n {{ .Release.Namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "Kouma: http://$NODE_IP:$NODE_PORT"

{{- else if eq .Values.dashboard.service.type "LoadBalancer" }}

Access Kouma via LoadBalancer:
  export LB_IP=$(kubectl get svc {{ $fullName }}-dashboard -n {{ .Release.Namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "Kouma: http://$LB_IP:{{ .Values.dashboard.service.port }}"

{{- else }}

Access the dashboard directly:
  kubectl port-forward svc/{{ $fullName }}-dashboard {{ .Values.dashboard.service.port }}:{{ .Values.dashboard.service.port }} -n {{ .Release.Namespace }}
  echo "Dashboard: http://localhost:{{ .Values.dashboard.service.port }}"

{{- end }}
